饼状图自动算百分比 #

const legend = {
  right: '5%',
  orient: 'vertical',
  formatter: function (name) {
    let data = [
      { value: 1048, name: 'aa' },
      { value: 735, name: 'bb' },
      { value: 580, name: 'cc' },
      { value: 484, name: 'dd' }
    ] // console.log(data, 'data')
    let total = 0
    let tarValue
    for (let i = 0; i < data.length; i++) {
      total += data[i].value
      if (data[i].name == name) {
        tarValue = data[i].value
      }
    }
    let v = tarValue + '人' //计算出百分比
    let p = Math.round((tarValue / total) * 100) + '%'
    return `${name}  ${v}  ${p}` //name是名称，v是数值
  }
}
